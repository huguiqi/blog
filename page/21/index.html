//匹配电话号码(手机号以13， 15，18开头，八个 \d
数字字符)\
BOOL isMatch = [ “15910514636” isMatch:RX( “\^((13[0-9])|(15[\^4,\\D])|(18[0,0-9]))\\d{8}\$“)];\
NSLog( “%d”, isMatch);\
\
//匹配邮箱\
